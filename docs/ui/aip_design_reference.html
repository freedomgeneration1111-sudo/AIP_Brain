def aip_card(title, artifact_id, body, sources=None, status='Generated',
             on_approve=None, on_reject=None):
    with ui.column().style(
        f'background:{C_SURFACE}; border:{BORDER.replace(C_INK40, C_RAISED)}; '
        f'border-radius:{R_LG}; padding:16px; gap:0; width:100%;'
    ):
        # Header row: title + status pill
        with ui.row().style('align-items:flex-start; justify-content:space-between; margin-bottom:10px; width:100%;'):
            with ui.column().style('gap:3px; flex:1;'):
                ui.label(title).style(f'font-size:13px; font-weight:500; color:{C_CREAM}; line-height:1.4;')
                ui.label(artifact_id).style(f'font-family:{F_MONO}; font-size:9px; color:{C_INK60};')
            status_pill(status)
        # Body
        ui.label(body).style(f'font-size:12px; color:{C_MUTED}; line-height:1.65; margin-bottom:12px;')
        # Sources
        if sources:
            ui.separator().style(f'background:{C_INK40}; height:0.5px; margin-bottom:10px;')
            ui.label(f'Sources · {len(sources)} retrieved').style(
                f'font-size:9px; letter-spacing:1.5px; color:{C_INK60}; text-transform:uppercase; margin-bottom:6px;'
            )
            for s in sources:
                ui.label(f'· {s}').style(f'font-size:10px; color:#4A6070; margin-bottom:4px;')
        # Actions
        with ui.row().style('gap:8px; margin-top:12px;'):
            ui.button('Approve', on_click=on_approve).style(btn_primary())
            ui.button('Reject', on_click=on_reject).style(btn_secondary())
            ui.button('View sources ↗').style(btn_ghost())

def status_pill(status: str) -> None:
    styles = {
        'Generated': ('#B8935A', '#8C6E3A', '#1A1200'),
        'Approved':  ('#4EAA7A', '#1E4030', '#0E1F17'),
        'Rejected':  ('#E07070', '#3A1E1E', '#1A0000'),
        'Idle':      ('#8FA8B8', '#2A3540', 'transparent'),
        'Active':    ('#B8935A', '#8C6E3A', '#1A0E00'),
    }
    fg, border, bg = styles.get(status, ('#8FA8B8', '#2A3540', 'transparent'))
    ui.label(status).style(
        f'display:inline-flex; align-items:center; font-size:10px; letter-spacing:.5px; '
        f'padding:3px 8px; border-radius:4px; border:0.5px solid {border}; '
        f'color:{fg}; background:{bg}; flex-shrink:0;'
    )

5 · Usage Rules — Non-Negotiable

✓ DO
Amber only for: primary CTA, active tab indicator, corpus center node, Beast active dot
Georgia serif only for: wordmark, display headings, nothing else
0.5px borders on all surface separations
Dark ground (#0E0E0F) as the primary application environment
Monospace for all IDs, timestamps, status bar text
Small-caps with 2px letter-spacing for all section labels
✗ NEVER
Gradients, drop shadows, or blur effects anywhere
Pure black (#000) or pure white (#FFF)
More than two accent colors visible in one view
Amber for decorative purposes — it signals action only
Rounded corners larger than 8px
Border width other than 0.5px for structural borders
Inter, Roboto, or any custom webfont (system-sans only for body)
⚠ NiceGUI-specific
NiceGUI renders buttons with its own default styling. Always chain .style() and .props('flat') or override with .classes('') to strip Quasar defaults before applying AIP styles. Test every button in the running app — NiceGUI's Quasar layer can override inline styles unexpectedly.
